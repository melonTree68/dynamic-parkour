\section{Motivation}

\begin{frame}{Problem}
    \begin{columns}[T,onlytextwidth]
        \column{0.48\textwidth}
        \textbf{Static parkour}
        \begin{itemize}
            \setlength{\itemsep}{0.45em}
            \item Obstacles are fixed after terrain generation.
            \item The policy can rely on local height scans and waypoint geometry.
            \item The original two-stage pipeline works well on A1 parkour.
        \end{itemize}

        \column{0.48\textwidth}
        \textbf{Dynamic parkour}
        \begin{itemize}
            \setlength{\itemsep}{0.45em}
            \item Obstacle pose and support surfaces change over time.
            \item Timing becomes part of foot placement and takeoff selection.
            \item A deployable policy must infer time-varying obstacle state.
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Project Question}
    \begin{columns}[T,onlytextwidth]
        \column{0.52\textwidth}
        \textbf{Can the original privileged-to-depth pipeline survive moving obstacles?}

        \vspace{0.8em}
        We study this by keeping the A1 parkour control stack, then changing the terrain process and latent recovery path around it.

        \column{0.42\textwidth}
        \begin{block}{Contributions}
            \begin{itemize}
                \setlength{\itemsep}{0.35em}
                \item Dynamic-obstacle task infrastructure
                \item Static-expert DAgger pretraining
                \item Dynamic environment latent augmentation
                \item Mixed-terrain latent suppression
                \item Optional depth encoder latent loss
            \end{itemize}
        \end{block}
    \end{columns}
\end{frame}

\section{Baseline Pipeline}

\begin{frame}{Original Training Pipeline}
    \begin{columns}[T,onlytextwidth]
        \column{0.47\textwidth}
        \textbf{Phase 1: base RL}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item PPO trains a privileged policy in simulation.
            \item Actor input includes proprioception, height scan latent, explicit privileged state, and domain-randomization latent.
            \item A history encoder learns to replace the privileged latent at deployment-like inference time.
        \end{itemize}

        \column{0.47\textwidth}
        \textbf{Phase 2: camera distillation}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item A depth encoder predicts scan latent and yaw correction from depth and proprioception.
            \item A student actor imitates the teacher action while running with depth-derived latent input.
            \item The original active loss is action/yaw distillation; scan-latent matching is optional in our extension.
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Curriculum and Evaluation}
    \begin{columns}[T,onlytextwidth]
        \column{0.50\textwidth}
        \textbf{Terrain curriculum}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item 10 difficulty rows by 40 terrain columns.
            \item Each tile is 18 m by 4 m with eight navigation goals.
            \item Reset curriculum moves robots to harder or easier rows based on progress.
        \end{itemize}

        \column{0.44\textwidth}
        \textbf{Main metrics}

                Length     & Survival time   \\

                \bottomrule
            \end{tabular}
        \end{table}
    \end{columns}
\end{frame}

\section{Dynamic Task}

\begin{frame}{\texttt{a1\_dynamic}}
    \begin{columns}[T,onlytextwidth]
        \column{0.49\textwidth}
        \textbf{Design}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Registered separately from \texttt{a1}.
            \item Uses \texttt{DynamicLeggedRobot} with the A1 parkour config inheritance path.
            \item Keeps the eight-goal course interface and height-scan observation shape.
            \item Replaces static terrain families with dynamic obstacle families.
        \end{itemize}

        \column{0.45\textwidth}
        \textbf{Dynamic families}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Moving hurdles
            \item Moving gap platforms
            \item Tilting pads
            \item Vertically moving steps
            \item Mixed dynamic-demo tracks
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Runtime Infrastructure}
    \begin{columns}[T,onlytextwidth]
        \column{0.52\textwidth}
        \textbf{Static terrain metadata}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Terrain generation stores dynamic family, difficulty, obstacle specs, motion types, motion groups, and goal groups.
            \item Six obstacle groups and two actor slots per group give twelve additional actors per environment.
        \end{itemize}

        \column{0.42\textwidth}
        \textbf{Isaac Gym actors}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Dynamic obstacles are gravity-disabled box actors.
            \item Scripted sinusoidal motion updates actor root states before physics.
            \item Height scans analytically overlay current actor top surfaces.
            \item Gap-related goals move with their support surfaces.
        \end{itemize}
    \end{columns}
\end{frame}

\section{Pipeline Improvements}

\begin{frame}{DAgger-Style Imitation Pretraining}
    \begin{columns}[T,onlytextwidth]
        \column{0.46\textwidth}
        \textbf{Motivation}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Dynamic parkour is hard to learn from scratch.
            \item A static A1 expert provides a strong but shifted behavior prior.
            \item DAgger labels both expert-driven and student-driven states to reduce covariate shift.
        \end{itemize}

        \column{0.48\textwidth}
        \textbf{Training target}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Privileged actor action imitation
            \item History-branch action imitation
            \item Explicit privileged-state estimator loss
            \item Dynamic latent ROA loss for configured families
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{DAgger Pseudocode}
    \small
    \begin{algorithmic}[1]
        \Require static expert $\pi_E$, dynamic env, student $\pi_\theta$
        \State Initialize teacher replay $B_E$ and student replay $B_S$
        \For{$k=0,\ldots,K-1$}
        \If{$k$ is the warmup iteration}
        \State Roll out $\pi_E$ and store $(o,\pi_E(o))$ in $B_E$
        \Else
        \State Roll out $\pi_E$ into $B_E$
        \State Roll out $\pi_\theta$ into $B_S$, labeling states with $\pi_E$
        \EndIf
        \State Sample a balanced minibatch from $B_E$ and $B_S$
        \State Update $\pi_\theta$ with action, history-action, estimator, and dynamic-latent losses
        \State Save a normal checkpoint for RL fine-tuning
        \EndFor
    \end{algorithmic}
\end{frame}

\begin{frame}{Dynamic Environment Latent}
    \begin{columns}[T,onlytextwidth]
        \column{0.51\textwidth}
        \textbf{Privileged label}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item 2 upcoming obstacle groups.
            \item 15 features per group.
            \item Includes validity, family bits, relative position, obstacle value, velocity, acceleration, phase, amplitude, and frequency.
        \end{itemize}

        \column{0.43\textwidth}
        \textbf{Recovery modes}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item \textbf{ROA}: history encoder predicts dynamic latent from proprioceptive history.
            \item \textbf{Teacher-student}: depth encoder predicts dynamic latent during camera distillation.
            \item \textbf{Hybrid}: choose recovery path per obstacle family.
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Depth Encoder Loss}
    \begin{columns}[T,onlytextwidth]
        \column{0.49\textwidth}
        \textbf{Optional scan-latent matching}
        \[
            L_{\mathrm{scan}}
            =
            \left\|\operatorname{sg}(z_{\mathrm{scan}}) - z_{\mathrm{depth}}\right\|_2
        \]
        \vspace{-0.5em}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Enabled by \texttt{--train\_depth\_encoder\_loss}.
            \item Disabled by default for compatibility.
        \end{itemize}

        \column{0.45\textwidth}
        \textbf{Integrated update}
        \[
            L =
            L_{\mathrm{action}}
            + L_{\mathrm{yaw}}
            + w_{\mathrm{dyn}}L_{\mathrm{dyn}}
            + L_{\mathrm{scan}}
        \]
        \vspace{-0.5em}
        The loss is added to the same depth actor optimizer step, avoiding a second backward pass through shared depth-encoder tensors.
    \end{columns}
\end{frame}

\section{Conclusion}

\begin{frame}{Takeaways}
    \begin{columns}[T,onlytextwidth]
        \column{0.49\textwidth}
        \textbf{What worked}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Dynamic task infrastructure supports moving obstacles without changing the high-level A1 parkour interface.
            \item Static-expert DAgger gives a better dynamic fine-tuning starting point.
            \item Dynamic latent augmentation gives an explicit handle for testing what obstacle state is recoverable.
        \end{itemize}

        \column{0.45\textwidth}
        \textbf{Final message}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Dynamic obstacles expose a clear limitation of the original static-terrain pipeline.
            \item Static-expert imitation gives dynamic RL a stronger initialization.
            \item Dynamic state recovery should be family-aware: explicit where useful, implicit or suppressed where not.
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Future Directions}
    \begin{itemize}
        \setlength{\itemsep}{0.55em}
        \item Evaluate by obstacle family and failure mode rather than only aggregate reward.
        \item Add targeted timing metrics such as takeoff error and landing stability.
        \item Stress-test depth recovery under partial observability, latency, and low frame rate.
        \item Use dynamic latent labels as an analysis tool even when the best policy uses implicit recovery.
    \end{itemize}
\end{frame}
