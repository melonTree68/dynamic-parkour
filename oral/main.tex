\section{Motivation}

\begin{frame}{Problem}
    \textbf{Static parkour}
    \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item Obstacles are fixed after terrain generation.
        \item The policy can rely on local height scans and waypoint geometry.
        \item The original two-stage pipeline works well on A1 parkour.
    \end{itemize}

    \vspace{0.8em}
    \textbf{Dynamic parkour}
    \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item Obstacle pose and support surfaces change over time.
        \item Timing becomes part of foot placement and takeoff selection.
        \item A deployable policy must infer time-varying obstacle state.
    \end{itemize}
\end{frame}

\begin{frame}{Project Question}
    \textbf{Can the original privileged-to-depth pipeline survive moving obstacles?}

    \vspace{0.4em}
    We keep the A1 parkour control stack, then change the terrain process and latent recovery path around it.

    \vspace{0.8em}
    \textbf{Contributions}
    \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item Dynamic-obstacle task infrastructure
        \item Static-expert DAgger pretraining
        \item Dynamic environment latent augmentation
        \item Mixed-terrain latent suppression
        % \item Depth encoder latent loss
    \end{itemize}
\end{frame}

\begin{frame}{Related Work and Background}
    \begin{columns}[T,onlytextwidth]
        \column{0.48\textwidth}
        \textbf{Privileged learning for locomotion}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Train a teacher with simulator-only terrain and dynamics information.
            \item Distill deployment behavior to history or depth observations.
            \item Robustness depends on what latent state can be recovered.
        \end{itemize}

        \column{0.47\textwidth}
        \textbf{Gap in this project}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Existing parkour benchmark focuses on static obstacle geometry.
            \item Moving supports add timing and partial-observability failures.
            \item We test whether latent recovery should be proprioceptive, visual, or family-specific.
        \end{itemize}
    \end{columns}
\end{frame}

\section{Baseline Pipeline}

\begin{frame}{Original Training Pipeline}
    \begin{columns}[T,onlytextwidth]
        \column{0.52\textwidth}
        \includegraphics[width=\linewidth,height=0.70\textheight,keepaspectratio]{reference/original_paper/method_compressed.pdf}

        \column{0.43\textwidth}
        \footnotesize
        \textbf{Phase 1: base RL}
        \begin{itemize}
            \setlength{\itemsep}{0.25em}
            \item PPO trains a privileged teacher in simulation.
            \item Actor sees proprioception, scan latent, explicit privileged state, and privileged latent.
        \end{itemize}

        \vspace{0.5em}
        \textbf{Phase 2: camera distillation}
        \begin{itemize}
            \setlength{\itemsep}{0.25em}
            \item Depth encoder predicts scan latent and yaw correction.
            \item Student actor imitates teacher action with depth-derived latent input.
        \end{itemize}
        \textbf{Loss:} action imitation plus yaw prediction.
    \end{columns}
\end{frame}

\begin{frame}{Original Latent Recovery Modes}
    \begin{columns}[T,onlytextwidth]
        \column{0.48\textwidth}
        \textbf{ROA-style history recovery}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item During base RL, the teacher uses privileged latent variables.
            \item A history encoder learns to match the privileged latent from proprioceptive history.
            \item This is suitable for state that can be inferred from recent robot motion.
        \end{itemize}

        \column{0.48\textwidth}
        \textbf{Teacher-student depth recovery}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item During camera distillation, the teacher acts with terrain scan information.
            \item The depth student predicts scan latent and yaw correction.
            \item This is suitable for obstacle state that must be seen before contact.
        \end{itemize}
    \end{columns}
\end{frame}

    \begin{figure}
        \centering
        \includegraphics[width=0.72\linewidth]{curriculum_tiles_labeled.png}
        \vspace{-1.0em}
        \caption{Curriculum grid concept: columns select terrain families; rows increase difficulty.}

\begin{frame}{Static Terrain Difficulty}
    \begin{itemize}
        \setlength{\itemsep}{0.4em}
        \item The simulator builds a 10-row by 40-column terrain grid; each tile is 18 m by 4 m with eight navigation goals.
        \item Row index defines difficulty from 0 to 1; column index samples the terrain family from configured proportions.
        \item Each terrain generator constructs a heightfield tile, places waypoint goals, adds roughness, and then the global heightfield is converted to a triangle mesh.
        \item Difficulty changes obstacle geometry: ramp length/height, hurdle thickness/height, step height, gap width, lateral ranges, and platform spacing.
        \item Actual obstacle values are sampled uniformly inside difficulty-dependent ranges, so policies see a distribution rather than one deterministic course per row.
        \item Reset curriculum moves robots to harder or easier rows based on progress.
    \end{itemize}
\end{frame}

\section{Dynamic Task}

\begin{frame}{\texttt{a1\_dynamic}}
    \begin{columns}[T,onlytextwidth]
        \column{0.49\textwidth}
        \textbf{Design}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Keeps the eight-goal course interface and height-scan observation shape.
            \item Replaces static terrain families with dynamic obstacle families.
            \item Adds a unified parameter-tuning interface for moving obstacle geometry and motion.
        \end{itemize}

        \column{0.45\textwidth}
        \textbf{Dynamic families}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Moving hurdles
            \item Moving gap platforms
            \item Tilting pads
            \item Vertically moving steps
            \item Mixed dynamic-demo tracks
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Runtime Infrastructure}
    \textbf{Isaac Gym actors}
    \begin{itemize}
        \setlength{\itemsep}{0.45em}
        \item Dynamic obstacles are gravity-disabled box actors.
        \item Scripted sinusoidal motion updates actor root states before physics.
        \item Height scans analytically overlay current actor top surfaces.
        \item Gap-related goals move with their support surfaces.
    \end{itemize}
\end{frame}

\begin{frame}{Dynamic Terrain Construction}
    \begin{itemize}
        \setlength{\itemsep}{0.38em}
        \item Dynamic terrain keeps the static grid, tile size, waypoint count, roughness, and curriculum row logic.
        \item The added part is obstacle metadata: family, current difficulty, actor dimensions, motion type, motion group, and which goals move with support surfaces.
        \item Hurdles are moving bars; gaps are moving takeoff/landing platform pairs; tilted pads are rolling support boxes; steps are vertically moving blocks.
        \item Pure gap tiles synchronize gap motion to avoid accidental extra openings; demo tiles combine hurdle, step, gap, and tilted-pad groups.
        \item As in static terrain, actual per-obstacle values are sampled uniformly from ranges determined by difficulty.
    \end{itemize}
\end{frame}

\begin{frame}{Dynamic Difficulty Parameters}
    \scriptsize
    \begin{table}
        \centering
        \caption{Dynamic-only parameters interpolated by difficulty before uniform sampling.}
        \begin{tabular}{lll}
            \toprule
            Family & Geometry range & Motion range \\
            \midrule
            Hurdle & height $[0.10,0.15]\rightarrow[0.20,0.40]$ m & amplitude $0.02\rightarrow0.12$ m; period $5.0\rightarrow2.5$ s \\
            Gap & size $0.10\rightarrow0.80$ m; moving platform pair & amplitude $0.05\rightarrow0.30$ m; period $4.0\rightarrow2.0$ s \\
            Tilted pad & pad size fixed; lateral range expanded & roll amplitude $0.10\rightarrow0.45$ rad; period $4.5\rightarrow2.5$ s \\
            Step & increment $[0.10,0.10]\rightarrow[0.40,0.50]$ m & amplitude $0.02\rightarrow0.12$ m; period $4.5\rightarrow2.5$ s \\
            \bottomrule
        \end{tabular}
    \end{table}
    Amplitudes are sampled as a high fraction of the difficulty-scaled maximum; smaller periods mean faster obstacle motion.
\end{frame}

\section{Pipeline Improvements}

\begin{frame}{DAgger-Style Imitation Pretraining}
    \begin{columns}[T,onlytextwidth]
        \column{0.46\textwidth}
        \textbf{Motivation}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Dynamic parkour is hard to learn from scratch.
            \item A static A1 expert provides a strong but shifted behavior prior.
            \item DAgger labels both expert-driven and student-driven states to reduce covariate shift.
        \end{itemize}

        \column{0.48\textwidth}
        \textbf{Training target}
        \begin{itemize}
            \setlength{\itemsep}{0.35em}
            \item Privileged actor action imitation
            \item History-branch action imitation
            \item Explicit privileged-state estimator loss
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{DAgger Pseudocode}
    \small
    \begin{algorithmic}[1]
        \Require static expert $\pi_E$, dynamic env
        \State Initialize student $\pi_\theta$
        \State Initialize teacher replay buffer $B_E$ and student replay buffer $B_S$
        \For{$k=0,\ldots,K-1$}
        \If{$k$ is the BC warmup iteration}
        \State Roll out $\pi_E$ and store $(o,\pi_E(o))$ in $B_E$
        \Else
        \State Roll out $\pi_E$ into $B_E$
        \State Roll out $\pi_\theta$ into $B_S$, labeling states with $\pi_E$
        \EndIf
        \State Sample a balanced minibatch from $B_E$ and $B_S$
        \State Update $\pi_\theta$ with action, history-action, and estimator losses
        \EndFor
        \State \Return $\pi_\theta$ for base RL fine-tuning
    \end{algorithmic}
\end{frame}

\begin{frame}{Environment Latent Augmentation}
    \textbf{What is augmented}
    \begin{itemize}
        \setlength{\itemsep}{0.3em}
        \item Add dynamic obstacle state to the environment latent.
        \item Encode the next two obstacle groups.
        \item Each group has 15 features.
    \end{itemize}
    \vspace{1.0em}
    \textbf{Why redundant}
    \begin{itemize}
        \setlength{\itemsep}{0.3em}
        \item The representation is not the minimal physical state.
        \item It repeats type, geometry, phase, and derivative information.
        \item Lower-dimensional labels were expected to be too weak for learning a useful recovery signal.
    \end{itemize}
\end{frame}

\begin{frame}{Dynamic Obstacle State Features}
    \scriptsize
    \begin{table}
        \centering
        \caption{15 features per upcoming obstacle group.}
        \begin{tabular}{ll}
            \toprule
            Category & Features \\
            \midrule
            Validity and type & valid, hurdle bit, gap bit, step bit, tilted-pad bit \\
            Relative geometry & rel-$x$ slot 0, rel-$x$ slot 1, rel-$y$, primary value \\
            Motion state & value velocity, value acceleration, $\sin(\text{phase})$, $\cos(\text{phase})$ \\
            Motion parameters & amplitude, angular frequency \\
            \bottomrule
        \end{tabular}
    \end{table}
    The primary value is family-specific: hurdle top height, effective gap width, step top height, or tilted-pad roll angle.
\end{frame}

\begin{frame}{Per-Family Interpretation}
    \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item \textbf{Hurdle}: imitation pretraining is strongest; hurdle motion is small, while collision artifacts dominate the failure mode.
        \item \textbf{Tilted pads}: augmentation hurts. The robot can aim for the rotation axis, so roll angle is less useful and becomes a training distraction.
        \item \textbf{Gap}: teacher-student is strongest because takeoff timing and landing recovery require visual information.
        \item \textbf{Step}: dynamic step is not learnable in current Isaac Gym physics; collisions with the step actor push the robot back strongly and prevent reward collection.
    \end{itemize}
\end{frame}

        \begin{itemize}
            \setlength{\itemsep}{0.35em}

            \item Gap state remains routed through teacher-student recovery.
        \end{itemize}
    \end{columns}

\section{Conclusion}

\begin{frame}{Takeaways}
    \textbf{What worked}
    \begin{itemize}
        \setlength{\itemsep}{0.45em}
        \item Dynamic task infrastructure supports moving obstacles through a unified parameter-tuning interface.
        \item Static-expert DAgger gives a better dynamic fine-tuning starting point.
        \item Dynamic latent augmentation gives an explicit handle for testing what obstacle state is recoverable.
        % \item Adding depth encoder loss makes distillation converge faster and reach better performance.
    \end{itemize}
\end{frame}

\begin{frame}{Future Directions}
    \begin{itemize}
        \setlength{\itemsep}{0.55em}
        \item Turn targeted timing signals, such as takeoff error and landing stability, into both metrics and losses.
        \item Stress-test depth recovery under partial observability, latency, and low frame rate.
        \item Use dynamic latent labels as an analysis tool even when the best policy uses implicit recovery.
    \end{itemize}
\end{frame}
